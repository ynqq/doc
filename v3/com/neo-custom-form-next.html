                                                    # 组件说明

                                                    可拖拽自定义表单

                                                    # 安装方式

                                                    pnpm install @neotrident/neo-custom-form-next --save

                                                    npm install @neotrident/neo-custom-form-next --save

                                                    yarn add @neotrident/neo-custom-form-next

                                                    # 使用方法

                                                    1. 在 main.ts 里面引入
                                                    import NeoCustomFormNext from '@neotrident/neo-custom-form-next';
import '@neotrident/neo-custom-form-next/dist/style.css';
...
app.use(NeoCustomFormNext)
                                                    1. 类型支持
                                                    export interface GlobalComponents {
    NeoCustomFormNext: typeof import('@neotrident/neo-custom-form-next/dist/packages/CustomForm/index.vue')['default'];
}
                                                    1. 使用案例
                                                    <NeoCustomFormNext
  v-model:value="data"
  :header="getHeader"
  :upload-action="webConfig.api.sip + '/api/File/upload'"
>
  <template #preset-fields>
    <div></div>
  </template>
</NeoCustomFormNext>

  isTem?: boolean;
  id: number;
  type:
    | "input"
    | "textarea"
    | "select"
    | "multiple"
    | "radio"
    | "checkbox"
    | "date"
    | "inputNumber"
    | "email"
    | "link"
    | "attachment"
    | "container2"
    | "container3";
  title: string;
  model?: string | string[] | number | number[] | undefined;
  icon?: any;
  placeholder?: string;
  action?: string;
  nameMapping?: string;
  urlMapping?: string;
  fileName?: string;
  defaultList?: any[];
  value?: ModalValueProps[];
  format?: any;
  formatOptions?: any[];
  append?: string;
  prepend?: string;
  hasSuffix?: boolean;
  suffixOptions?: boolean[];
  columns?: ContainerColumnsProp;
}

export interface ContainerColumnsProp {
  l: ModalItemProps[];
  m?: ModalItemProps[];

                                                    # uploadAction

                                                    上传地址

                                                    • 类型: string
                                                    • 默认值: ''

                                                    # height

                                                    组件高度

                                                    • 类型: string
                                                    • 默认值：'100%'

                                                    # mode

                                                    组件模式

                                                    • 类型:
                                                      • design: 可以拖动
                                                      • edit: 不能拖动但是可以对表单进行写入
                                                      • view: 只能看，不能拖动，不能写入
                                                    • 默认值: 'design'

                                                    # lang

                                                    语言类型

                                                    • 类型: "zh-CN" | "en-US"
                                                    • 默认值: "zh-CN"

                                                    # langPackage

                                                    语言包,可以用来替换默认的语言包

                                                    • 类型: object
                                                    • 默认值: {}

                                                    # header

                                                    请求接口时的 header

                                                    • 类型: () => AxiosHeader
                                                    • 默认值: () => ({})

                                                    # uploadInfo

                                                    文件上传信息

                                                    • 类型: UploadInfoProps

                                                      export interface UploadInfoProps {
  statusKey: string; // 上传接口返回的状态 key 名
  successValue: number; // 成功时的状态值
  uploadFileName: string; // 上传文件的名称(不传默认为文件名称)
}

                                                    • 默认值:

                                                      {
    statusKey: "Status",
    successValue: 1,
    uploadFileName: "",
}

                                                    # hasColon

                                                    是否在表单 label 后面加入: 例如: 姓名:

                                                    • 类型: boolean
                                                    • 默认值: false

                                                    # domain

                                                    文件预览时的域名

                                                    • 类型: string
                                                    • 默认值: ''
